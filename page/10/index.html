    onEnter(args) {
        //在函数调用前产生的回调,在这个函数中可以处理函数参数的相关内容.
        //被Hook的函数参数内容是以数组的方式存储在args中.

    },
    onLeave(retval) {
        //在被Hook的目标函数执行完成后执行的函数.
        //被Hook的函数返回值用retval变量来表示.

    }
});
